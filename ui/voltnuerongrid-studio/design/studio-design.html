-- Analyze high-value users with recent orders SELECT u.id, u.email, COUNT(o.id) AS order_count, SUM(o.total_amount) AS lifetime_value, MAX(o.created_at) AS last_order_at FROM users u JOIN orders o ON o.user_id = u.id WHERE u.is_active = TRUE AND o.created_at > NOW() - INTERVAL '30 days' GROUP BY u.id, u.email HAVING SUM(o.total_amount) > 1000 ORDER BY lifetime_value DESC LIMIT 100;
